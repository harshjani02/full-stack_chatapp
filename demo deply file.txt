# in backend ::   package.json : Defines what dependencies the project needs.
#  package-lock.json: Locks the exact versions of those dependencies so every environment installs the same packages.


# Q: Why use npm ci instead of npm install in CI/CD?

# Answer:

# npm ci performs a clean installation using exact dependency versions from package-lock.json. It is faster, reproducible, and prevents unexpected dependency changes, making it ideal for CI/CD pipelines such as GitHub Actions, Jenkins, and Kubernetes deployments.

# For DevOps work, remember this simple rule:

# Local Development  → npm install
# CI/CD Pipeline     → npm ci

# Why use ${{ github.sha }}  [Secure Hash Algorithm] instead of latest?

# ${{ github.sha }} creates a unique Docker image tag for every commit, making deployments traceable, reproducible, and easy to roll back. The latest tag changes over time and does not identify which exact version of the code is running in production.

name : Full-stack deployment

on:
  push:
    branches:
      - main

env:
  IMAGE_NAME: fullstack-app

jobs:
  test:
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-node@v4
        with:
          node-version: 20
          cache: 'npm'
          cache-dependency-path: backend/package-lock.json
      - run: cd backend && npm ci
      

  docker:
    needs: test

    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v4 

      - name: Login Dockerhub
        uses: docker/login-action@v3
        with:
          username: ${{ secrets.DOCKER_USERNAME }}
          password: ${{ secrets.DOCKER_PASSWORD }}

      - name: Build IMAGE
        run : docker build -t $IMAGE_NAME .

      - name: Tag Image 
        run: docker tag $IMAGE_NAME ${{ secrets.DOCKER_USERNAME }}/$IMAGE_NAME:${{ github.sha }}   

      - name: Push Image
        run: docker push {{ secrets.DOCKER_USERNAME }}/$IMAGE_NAME:${{ github.sha }}

  deploy:
    needs: docker
    runs-on: ubuntu-latest

    environment:
      name: production

    steps:
      - name: Deploy to Server
        run: |
          echo "Deploying application"
